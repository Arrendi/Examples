sigr formats significance tests.

library('sigr')
d <- data.frame(x=1:5)
set.seed(353525)
d$y <- 2*d$x + rnorm(nrow(d))
model <- lm(y~x, d=d)
d$pred <- predict(model, newdata = d)

summary(model)
